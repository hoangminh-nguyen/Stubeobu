create table Account
(
	username varchar(10),
	pass varchar(20),
	constraint PK_Ac primary key (username)
);

create table Student
(
	student_id char(9),
	lastname nvarchar(30),
	firstname nvarchar(15),
    gender varchar(3),
	dob char(10),
	constraint PK_St primary key (student_id)
);

create table Teacher
(
	teacher_id char(4),
	lastname nvarchar(30),
	firstname nvarchar(15),
    gender varchar(3),
	dob char(10),
	constraint PK_Te primary key (teacher_id)
);
create table Course_info
(
	course_id char(8),
	course_name nvarchar(40),
	constraint PK_Co_info primary key (course_id)
);
create table Course
(
	course_id char(8),
	number int,
	sem_id int,
	room char(5),
	teacher_id char(4),
	constraint PK_Co primary key (course_id,number,sem_id)
);
create table Semester
(
	sem_id int,
	semester int,
	years int,
	constraint PK_Se primary key (sem_id)
);
create table Student_Course
(
	student_id char(9),
	course_id char(8),
	number int,
	sem_id int,
	midterm double,
	final double,
	constraint PK_St_Co primary key (student_id, course_id,number,sem_id)
);
create table Timetable
(
	course_id char(8),
	number int,
	sem_id int,
	week_day int, 
	period_no int,
	constraint PK_Ti primary key (course_id,number,sem_id,week_day,period_no)
);

alter table Student
	add constraint FK_St_Ac foreign key (student_id) references Account(username);
alter table Teacher
	add constraint FK_Te_Ac foreign key (teacher_id) references Account(username);
alter table Course
	add constraint FK_Co_Se foreign key (sem_id) references Semester(sem_id),
    add constraint FK_Co_CoInfo foreign key (course_id) references Course_info(course_id);
alter table Course
	add constraint FK_Co_Te foreign key (teacher_id) references Teacher(teacher_id);
alter table Student_Course
	add constraint FK_StCo_St foreign key (student_id) references Student(student_id),
    add constraint FK_StCo_Co foreign key (course_id, number, sem_id) references Course(course_id, number, sem_id);
alter table Timetable
	add constraint FK_Ti_Co foreign key (course_id, number, sem_id) references Course(course_id, number, sem_id);

SELECT stc.course_id, ci.course_name, CONCAT(te.lastname," ",te.firstname) as fullname, se.semester, se.years, co.room, ti.week_day, ti.period_no
FROM Student_Course stc join Course co on (stc.course_id = co.course_id and stc.number = co.number and stc.sem_id = co.sem_id) 
	join Course_info ci on (stc.course_id = ci.course_id) 
	join teacher te on (co.teacher_id = te.teacher_id)
	join semester se on (stc.sem_id = se.sem_id)
    join Timetable ti on (ti.course_id = co.course_id and ti.number = co.number and ti.sem_id = co.sem_id) 
WHERE student_id = 'S18127202';

SELECT ti.week_day, ti.period_no 
FROM Student_Course stc join Timetable ti on (stc.course_id = ti.course_id and stc.number = ti.number and stc.sem_id = ti.sem_id) 
WHERE stc.student_id = 'S18127202' and stc.course_id = 'CSC00001';
